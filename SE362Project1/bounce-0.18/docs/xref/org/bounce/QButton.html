179 	/***
180 	 * Paints the QButton's border, this border needs to be selected from all 
181 	 * component state specific borders.
182 	 *
183 	 * @param g the graphics.
184 	 */
185 	protected void paintBorder(Graphics g) {
186 	    ButtonModel model	= getModel();
187 	    Border border 		= getBorder();
188 	    Border tmpBorder	= null;
189 
190 		// Disabled
191 	    if( !model.isEnabled()) {
192 	    	if( model.isSelected()) {
193 	    		tmpBorder = getDisabledSelectedBorder();
194 	    	} else {
195 	    		tmpBorder = getDisabledBorder();
196 	    	}
197 
198     	// Pressed
199 	    } else if( model.isPressed() && model.isArmed()) {
200 	    	tmpBorder = getPressedBorder();
201 
202 	    // In Focus but not pressed.
203 	    } else if( hasFocus()) {
204 	    	if( model.isSelected()) {
205 	    		tmpBorder = getFocusedSelectedBorder();
206 	    	} else {
207 		    	tmpBorder = getFocusedBorder();
208 	    	}
209 
210 	    // Rollover but not pressed or in focus.  
211 	    } else if( isRolloverEnabled() && model.isRollover()) {
212 	    	if( model.isSelected()) {
213 	    		tmpBorder = getRolloverSelectedBorder();
214 	    	} else {
215 	    		tmpBorder = getRolloverBorder();
216 	    	}
217 
218 	    // Selected but not pressed or in focus rollover.
219 	    } else if( model.isSelected()) {
220 	    	tmpBorder = getSelectedBorder();
221 	    }
222 
223 	    if( tmpBorder != null) {
224 	    	border = tmpBorder;
225 	    }
226 
227 	    if ( border != null) {
228 	        border.paintBorder(this, g, 0, 0, getWidth(), getHeight());
229 	    }
230 	}
