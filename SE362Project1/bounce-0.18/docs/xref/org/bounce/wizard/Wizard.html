208     private Color getGradientColor() {
209         Color color = UIManager.getColor("Button.focus");
210 
211         if (color == null) {
212             return (new JPanel()).getBackground();
213         }
214         
215         while (((color.getBlue() + color.getRed() + color.getGreen()) / 3) < 128) {
216             color = color.brighter();
217         }
218         
219         return color;
220     }

241     }
242     
243     public void backPage() {
244         WizardPage page = cards.selected();
245         setPage(page.getBack());
246     }
247 
248     public void nextPage() {
249         WizardPage page = cards.selected();
250         WizardPage next = page.getNext();
251 
252         next.setBack(page);
253         setPage(next);
254     }
255 
256     protected void setPage(WizardPage page) {
257         if (cards.selected() != null) {
258             cards.selected().removeWizardPageListener(this);
259         }
260 
261         cards.show(page);
262 
263         setTitle(page.getTitle());
264         setDescription(page.getDescription());
265 
266         backButton.setEnabled(page.getBack() != null);
267         if (page.getNext() != null) {
268             nextButton.setEnabled(true);
269             getRootPane().setDefaultButton(nextButton);
270         } else {
271             nextButton.setEnabled(false);
272             getRootPane().setDefaultButton(finishButton);
273         }
274 
275         page.addWizardPageListener(this);
276     }
277     
278     public WizardPage getPage() {
279         return cards.selected();
280     }
281     
282     public void pageChanged(WizardPageEvent event) {
283         setPage((WizardPage)event.getSource());
284     }
