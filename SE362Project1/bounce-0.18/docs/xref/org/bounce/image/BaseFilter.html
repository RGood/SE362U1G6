30  package org.bounce.image;
31  
32  import java.awt.image.RGBImageFilter;
33  
34  /***
35   * The base filter for all the image filters in the bounce package. 
36   * Contains a couple of handy Utility methods.
37   *
38   * @version	$Revision: 1.4 $, $Date: 2008/01/28 21:28:37 $
39   * @author Edwin Dankert <edankert@gmail.com>
40   */
41  public abstract class BaseFilter extends RGBImageFilter {
42  
43  	/***
44  	 * Gets the red value out of the rgb value.
45  	 *
46  	 * @param rgb the rgb value of the pixel.
47  	 *
48  	 * @return the red component value.
49  	 */
50  	protected int getRed( int rgb) {
51  		return (rgb >> 16) & 0xff;
52  	}
53  
54  	/***
55  	 * Gets the blue value out of the rgb value.
56  	 *
57  	 * @param rgb the rgb value of the pixel.
58  	 *
59  	 * @return the blue component value.
60  	 */
61  	protected int getBlue( int rgb) {
62  		return rgb & 0xff;
63  	}
64  
65  	/***
66  	 * Gets the green value out of the rgb value.
67  	 *
68  	 * @param rgb the rgb value of the pixel.
69  	 *
70  	 * @return the green component value.
71  	 */
72  	protected int getGreen( int rgb) {
73  		return (rgb >> 8) & 0xff;
74  	}
75  
76  	/***
77  	 * Returns the rgb value for the blue, red and green values.
78  	 *
79  	 * @param rgb the previous rgb value of the pixel, with alpha value.
80  	 * @param red the red component value of the pixel.
81  	 * @param green the green component value of the pixel.
82  	 * @param blue the blue component value of the pixel.
83  	 *
84  	 * @return the rgb value.
85  	 */
86  	protected int getRGB( int rgb, int red, int green, int blue) {
87  		return (rgb & 0xff000000) | (red << 16) | (green << 8) | (blue << 0);
88  	}
89  	
90  	/***
91  	 * Converts the red green and blue values to one value.
92  	 *
93  	 * @param rgb the rgb value of the pixel.
94  	 *
95  	 * @return a component value (between 0-255).
96  	 */
97  	protected int getNTSCValue( int rgb) {
98  		int component = (int)((0.30 * getRed( rgb) + 
99  		                     0.59 * getGreen( rgb) + 
100 		                     0.11 * getBlue( rgb)) / 3);
101 							 
102 		return component;
103 	}
104 }
