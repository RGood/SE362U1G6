30  package org.bounce.image;
31  
32  /***
33   * A filter to create an X% darker image.
34   *
35   * @version	$Revision: 1.4 $, $Date: 2008/01/28 21:28:37 $
36   * @author Edwin Dankert <edankert@gmail.com>
37   */
38  public class DarkerFilter extends BaseFilter {
39  	private int percent;
40  
41  	/***
42  	 * Constructs a Filter that can be used to construct a darker image.
43  	 *
44  	 * @param p the percentage of darkness in the range 0..100, where 100 is 
45  	 *        the darkest, and 0 is the lightest.
46  	 */
47  	public DarkerFilter( int p) {
48  	    percent = p;
49  
50  	    canFilterIndexColorModel = true;
51  	}
52  	
53  	/***
54  	 * Overrides <code>RGBImageFilter.filterRGB</code>.
55  	 *
56  	 * @param x the horizontal position of the pixel.
57  	 * @param y the vertical position of the pixel.
58  	 * @param rgb the color of the pixel.
59  	 *
60  	 * @return the new rgb value
61  	 */
62  	public int filterRGB( int x, int y, int rgb) {
63  		int red 	= getRed( rgb);
64  		int green	= getGreen( rgb);
65  		int blue	= getBlue( rgb);
66  
67  		red = red * (100 - percent) / 100;
68  		blue =  blue * (100 - percent) / 100;
69  		green = green *(100 - percent) / 100;
70  		
71  	    return getRGB( rgb, red, green, blue);
72  	}
73  }
