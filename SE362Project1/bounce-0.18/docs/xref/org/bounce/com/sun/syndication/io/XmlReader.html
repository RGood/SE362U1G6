34  /***
35   * Character stream that handles (or at least attemtps to) all the necessary Voodo to figure out
36   * the charset encoding of the XML document within the stream.
37   * <p>
38   * IMPORTANT: This class is not related in any way to the org.xml.sax.XMLReader. This one IS a
39   * character stream.
40   * <p>
41   * All this has to be done without consuming characters from the stream, if not the XML parser
42   * will not recognized the document as a valid XML. This is not 100% true, but it's close enough
43   * (UTF-8 BOM is not handled by all parsers right now, XmlReader handles it and things work in all
44   * parsers).
45   * <p>
46   * The XmlReader class handles the charset encoding of XML documents in Files, raw streams and
47   * HTTP streams by offering a wide set of constructors.
48   * <P>
49   * By default the charset encoding detection is lenient, the constructor with the lenient flag
50   * can be used for an script (following HTTP MIME and XML specifications).
51   * All this is nicely explained by Mark Pilgrim in his blog,
52   * <a href="http://diveintomark.org/archives/2004/02/13/xml-media-types">
53   * Determining the character encoding of a feed</a>.
54   * <p>
55   * @author Alejandro Abdelnur
56   *
57   */

103     /***
104      * Creates a Reader for a raw InputStream.
105      * <p>
106      * It follows the same logic used for files.
107      * <p>
108      * If lenient detection is indicated and the detection above fails as per specifications it then attempts
109      * the following:
110      * <p>
111      * If the content type was 'text/html' it replaces it with 'text/xml' and tries the detection again.
112      * <p>
113      * Else if the XML prolog had a charset encoding that encoding is used.
114      * <p>
115      * Else if the content type had a charset encoding that encoding is used.
116      * <p>
117      * Else 'UTF-8' is used.
118      * <p>
119      * If lenient detection is indicated an XmlReaderException is never thrown.
120      * <p>
121      * @param is InputStream to create a Reader from.
122      * @param lenient indicates if the charset encoding detection should be relaxed.
123      * @throws IOException thrown if there is a problem reading the stream.
124      * @throws XmlReaderException thrown if the charset encoding could not be determined according to the specs.
125      *
126      */
127     public XmlReader(InputStream is,boolean lenient) throws IOException, XmlReaderException {
128         try {
129             doRawStream(is,lenient);
130         } catch (XmlReaderException ex) {
131             if (!lenient) {
132                 throw ex;
133             }
134 
135             doLenientDetection(null,ex);
136         }
137     }

160      * Creates a Reader using the InputStream of a URLConnection.

163      * 'content-type' header in the fetched data it uses the same logic used for files.
164      * <p>
165      * If the URLConnection is a HTTP Url or there is a 'content-type' header in the fetched
166      * data it uses the same logic used for an InputStream with content-type.

175     public XmlReader(URLConnection conn) throws IOException {
176         boolean lenient = true;
177         if (conn instanceof HttpURLConnection) {
178             try {
179                 doHttpStream(conn.getInputStream(),conn.getContentType(),lenient);
180             }
181             catch (XmlReaderException ex) {
182                 doLenientDetection(conn.getContentType(),ex);
183             }
184         }
185         else
186         if (conn.getContentType()!=null) {
187             try {
188                 doHttpStream(conn.getInputStream(),conn.getContentType(),lenient);
189             }
190             catch (XmlReaderException ex) {
191                 doLenientDetection(conn.getContentType(),ex);
192             }
193         }
194         else {
195             try {
196                 doRawStream(conn.getInputStream(),lenient);
197             }
198             catch (XmlReaderException ex) {
199                 doLenientDetection(null,ex);
200             }
201         }
202     }

263     private void doLenientDetection(String httpContentType, XmlReaderException ex) throws IOException {
264         if (httpContentType!=null) {
265             if (httpContentType.startsWith("text/html")) {
266                 httpContentType = httpContentType.substring("text/html".length());
267                 httpContentType = "text/xml" + httpContentType;
268                 try {
269                     doHttpStream(ex.getInputStream(), httpContentType, true);
270                     ex = null;
271                 }
272                 catch (XmlReaderException ex2) {
273                     ex = ex2;
274                 }
275             }
276         }
277         if (ex!=null) {
278             String encoding = ex.getXmlEncoding();
279             if (encoding==null) {
280                 encoding = ex.getContentTypeEncoding();
281             }
282             if (encoding==null) {
283                 encoding = UTF_8;
284             }
285             prepareReader(ex.getInputStream(), encoding);
286         }
287     }

313     private void doRawStream(InputStream is,boolean lenient) throws IOException {
314         PushbackInputStream pis = new PushbackInputStream(is,PUSHBACK_MAX_SIZE);
315         String bomEnc = getBOMEncoding(pis);
316         String xmlGuessEnc =  getXMLGuessEncoding(pis);
317         String xmlEnc = getXmlProlog(pis,xmlGuessEnc);
318         String encoding = calculateRawEncoding(bomEnc, xmlGuessEnc, xmlEnc, pis);
319         prepareReader(pis,encoding);

338     // InputStream is passed for XmlReaderException creation only
339     private static String calculateRawEncoding(String bomEnc, String xmlGuessEnc, String xmlEnc, InputStream is) throws IOException {
340         String encoding;
341         if (bomEnc==null) {
342             if (xmlGuessEnc == null || xmlEnc == null) {
343                 encoding = UTF_8;
344             }
345             else
346             if (xmlEnc.equals(UTF_16) && (xmlGuessEnc.equals(UTF_16BE) || xmlGuessEnc.equals(UTF_16LE))) {
347                 encoding = xmlGuessEnc;
348             }
349             else {
350                 encoding = xmlEnc;
351             }
352         }
353         else
354         if (bomEnc.equals(UTF_8)) {
355             if (xmlGuessEnc!=null && !xmlGuessEnc.equals(UTF_8)) {
356                 throw new XmlReaderException(RAW_EX_1.format(new Object[]{bomEnc, xmlGuessEnc, xmlEnc}),
357                                              bomEnc, xmlGuessEnc, xmlEnc, is);
358             }
359             if (xmlEnc!=null && !xmlEnc.equals(UTF_8)) {
360                 throw new XmlReaderException(RAW_EX_1.format(new Object[]{bomEnc, xmlGuessEnc, xmlEnc}),
361                                              bomEnc, xmlGuessEnc, xmlEnc, is);
362             }
363             encoding = UTF_8;
364         }
365         else
366         if (bomEnc.equals(UTF_16BE) || bomEnc.equals(UTF_16LE)) {
367             if (xmlGuessEnc!=null && !xmlGuessEnc.equals(bomEnc)) {
368                 throw new IOException(RAW_EX_1.format(new Object[]{bomEnc, xmlGuessEnc, xmlEnc}));
369             }
370             if (xmlEnc!=null && !xmlEnc.equals(UTF_16) && !xmlEnc.equals(bomEnc)) {
371                 throw new XmlReaderException(RAW_EX_1.format(new Object[]{bomEnc, xmlGuessEnc, xmlEnc}),
372                                              bomEnc, xmlGuessEnc, xmlEnc, is);
373             }
374             encoding =bomEnc;
375         }
376         else {
377             throw new XmlReaderException(RAW_EX_2.format(new Object[]{bomEnc,xmlGuessEnc,xmlEnc}),
378                                          bomEnc,xmlGuessEnc,xmlEnc,is);
379         }
380         return encoding;
381     }

383     // InputStream is passed for XmlReaderException creation only
384     private static String calculateHttpEncoding(String cTMime, String cTEnc, String bomEnc, String xmlGuessEnc, String xmlEnc, InputStream is,boolean lenient) throws IOException {
385         String encoding;
386         if (lenient & xmlEnc!=null) {
387             encoding = xmlEnc;
388         }
389         else {
390             boolean appXml = isAppXml(cTMime);
391             boolean textXml = isTextXml(cTMime);
392             if (appXml || textXml) {
393                 if (cTEnc==null) {
394                     if (appXml) {
395                         encoding = calculateRawEncoding(bomEnc, xmlGuessEnc, xmlEnc, is);
396                     }
397                     else {
398                         encoding = US_ASCII;
399                     }
400                 }
401                 else
402                 if (bomEnc!=null && (cTEnc.equals(UTF_16BE) || cTEnc.equals(UTF_16LE))) {
403                     throw new XmlReaderException(HTTP_EX_1.format(new Object[]{cTMime,cTEnc,bomEnc,xmlGuessEnc,xmlEnc}),
404                                                  cTMime,cTEnc,bomEnc,xmlGuessEnc,xmlEnc,is);
405                 }
406                 else
407                 if (cTEnc.equals(UTF_16)) {
408                     if (bomEnc!=null && bomEnc.startsWith(UTF_16)) {
409                         encoding = bomEnc;
410                     }
411                     else {
412                         throw new XmlReaderException(HTTP_EX_2.format(new Object[]{cTMime,cTEnc,bomEnc,xmlGuessEnc,xmlEnc}),
413                                                      cTMime,cTEnc,bomEnc,xmlGuessEnc,xmlEnc,is);
414                     }
415                 }
416                 else {
417                     encoding = cTEnc;
418                 }
419             }
420             else {
421                 throw new XmlReaderException(HTTP_EX_3.format(new Object[]{cTMime,cTEnc,bomEnc,xmlGuessEnc,xmlEnc}),
422                                              cTMime,cTEnc,bomEnc,xmlGuessEnc,xmlEnc,is);
423             }
424         }
425         return encoding;
426     }

440     // returns charset parameter value, NULL if not present, NULL if httpContentType is NULL
441     private static String getContentTypeEncoding(String httpContentType) {
442         String encoding = null;
443         if (httpContentType!=null) {
444             int i = httpContentType.indexOf(";");
445             if (i>-1) {
446                 String postMime = httpContentType.substring(i+1);
447                 Matcher m = CHARSET_PATTERN.matcher(postMime);
448                 encoding = (m.find()) ? m.group(1) : null;
449                 encoding = (encoding!=null) ? encoding.toUpperCase() : null;
450             }
451         }
452         return encoding;
453     }

455     // returns the BOM in the stream, NULL if not present,
456     // if there was BOM the in the stream it is consumed
457     private static String getBOMEncoding(PushbackInputStream is) throws IOException {
458         String encoding = null;
459         int[] bytes = new int[3];
460         bytes[0] = is.read();
461         bytes[1] = is.read();
462         bytes[2] = is.read();
463 
464         if (bytes[0] == 0xFE && bytes[1] == 0xFF) {
465             encoding = UTF_16BE;
466             is.unread(bytes[2]);
467         }
468         else
469         if (bytes[0] == 0xFF && bytes[1] == 0xFE) {
470             encoding = UTF_16LE;
471             is.unread(bytes[2]);
472         }
473         else
474         if (bytes[0] == 0xEF && bytes[1] == 0xBB && bytes[2] == 0xBF) {
475             encoding = UTF_8;
476         }
477         else {
478             for (int i=bytes.length-1;i>=0;i--) {
479                 is.unread(bytes[i]);
480             }
481         }
482         return encoding;
483     }

485     // returns the best guess for the encoding by looking the first bytes of the stream, '<?'
486     private static String getXMLGuessEncoding(PushbackInputStream is) throws IOException {
487         String encoding = null;
488         int[] bytes = new int[4];
489         bytes[0] = is.read();
490         bytes[1] = is.read();
491         bytes[2] = is.read();
492         bytes[3] = is.read();
493         for (int i=bytes.length-1;i>=0;i--) {
494             is.unread(bytes[i]);
495         }
496 
497         if (bytes[0] == 0x00 && bytes[1] == 0x3C && bytes[2] == 0x00 && bytes[3] == 0x3F) {
498                 encoding = UTF_16BE;
499         }
500         else
501         if (bytes[0] == 0x3C && bytes[1] == 0x00 && bytes[2] == 0x3F && bytes[3] == 0x00) {
502                 encoding = UTF_16LE;
503         }
504         else
505         if (bytes[0] == 0x3C && bytes[1] == 0x3F && bytes[2] == 0x78 && bytes[3] == 0x6D) {
506             encoding = UTF_8;
507         }
508         return encoding;
509     }

513     // returns the encoding declared in the <?xml encoding=...?>,  NULL if none
514     private static String getXmlProlog(PushbackInputStream is,String guessedEnc) throws IOException {
515         String encoding = null;
516         if (guessedEnc!=null) {
517             byte[] bytes = new byte[PUSHBACK_MAX_SIZE];
518             int offset = 0;
519             int max = PUSHBACK_MAX_SIZE;
520             int c = is.read(bytes,offset,max);
521             while (c!=-1 && offset<PUSHBACK_MAX_SIZE) {
522                 offset += c;
523                 max -= c;
524                 c = is.read(bytes,offset,max);
525             }
526             int bytesRead = offset;
527             if (bytesRead>0) {
528                 is.unread(bytes,0,bytesRead);
529                 Reader reader = new InputStreamReader(new ByteArrayInputStream(bytes,0,bytesRead), guessedEnc);
530                 BufferedReader br = new BufferedReader(reader);
531                 String prolog = br.readLine();
532                 Matcher m = ENCODING_PATTERN.matcher(prolog);
533                 encoding = (m.find()) ? m.group(1).toUpperCase() : null;
534             }
535         }
536         return encoding;
537     }

539     // indicates if the MIME type belongs to the APPLICATION XML family
540     private static boolean isAppXml(String mime) {
541         return mime!=null &&
542                (mime.equals("application/xml") ||
543                 mime.equals("application/xml-dtd") ||
544                 mime.equals("application/xml-external-parsed-entity") ||
545                 (mime.startsWith("application/") && mime.endsWith("+xml")));
546     }

552                 mime.equals("text/xml-external-parsed-entity") ||
553                 (mime.startsWith("text/") && mime.endsWith("+xml")));
