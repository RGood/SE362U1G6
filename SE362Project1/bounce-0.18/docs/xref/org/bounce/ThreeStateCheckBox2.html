29  package org.bounce;
30  
31  import javax.swing.*;
32  import javax.swing.plaf.UIResource;
33  import java.awt.*;
34  import java.io.Serializable;
35  
36  /***
37   * Three state check box enhancement
38   *
39   * @date: 14.11.2007
40   * @author Edwin Dankert <edankert@gmail.com>, Evgeniy Smelik <sever@yandex.ru>
41   */
42  public class ThreeStateCheckBox2 extends JCheckBox {
43      private static final long serialVersionUID = -1842373249474550802L;
44  
45      public static final String NOT_SELECTED = "NOT_SELECTED";
46      public static final String SELECTED = "SELECTED";
47      public static final String DONT_CARE = "DONT_CARE";
48  
49      private String state = DONT_CARE;
50  
51      public ThreeStateCheckBox2(String aString) {
52          this(aString, DONT_CARE);
53      }
54  
55      public ThreeStateCheckBox2(String aString, String state) {
56          super(aString);
57  
58          Icon icon = UIManager.getIcon("CheckBox.icon");
59          ThreeStateCheckBoxIcon threeStateCheckBoxIcon = new ThreeStateCheckBoxIcon(icon);
60          setIcon(threeStateCheckBoxIcon);
61  
62          setSelected(state);
63      }
64  
65      public String getSelected() {
66          return state;
67      }
68  
69      public void setSelected(boolean selected) {
70          if (selected) {
71              state = SELECTED;
72          } else {
73              state = NOT_SELECTED;
74          }
75  
76          super.setSelected(selected);
77      }
78  
79      public void setSelected(String state) {
80          this.state = state;
81  
82          super.setSelected(SELECTED.equals(state) || DONT_CARE.equals(state));
83  
84          repaint();
85      }
86  
87      private static class ThreeStateCheckBoxIcon implements Icon, UIResource, Serializable {
88          private Icon icon;
89  
90          public ThreeStateCheckBoxIcon(Icon icon) {
91              this.icon = icon;
92          }
93  
94          public void paintIcon(Component c, Graphics g, int x, int y) {
95              ThreeStateCheckBox cb = (ThreeStateCheckBox) c;
96  
97              if (DONT_CARE.equals(cb.getSelected())) {
98                  Graphics2D g2d = ((Graphics2D) g);
99                  Composite composite = g2d.getComposite();
100                 AlphaComposite alphaComposite = AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 0.5f);
101                 g2d.setComposite(alphaComposite);
102                 icon.paintIcon(c, g, x, y);
103                 g2d.setComposite(composite);
104             } else {
105                 icon.paintIcon(c, g, x, y);
106             }
107         }
108 
109         public int getIconWidth() {
110             return icon.getIconWidth();
111         }
112 
113         public int getIconHeight() {
114             return icon.getIconWidth();
115         }
116     }
117 }
