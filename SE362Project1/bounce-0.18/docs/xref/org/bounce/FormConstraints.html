30  package org.bounce;
31  
32  /***
33   * A constraints object, used for the FormLayout, the constraints allow 
34   * for setting the components position, alignments and fill type.
35   *
36   * @version	$Revision: 1.4 $, $Date: 2008/01/28 21:28:37 $
37   * @author Edwin Dankert <edankert@gmail.com>
38   */
39  public class FormConstraints {
40  	/*** LEFT position */
41  	public static final int LEFT 	= 0;
42  	/*** BOTTOM position */
43  	public static final int BOTTOM	= 1;
44  	/*** RIGHT position */
45  	public static final int RIGHT	= 2;
46  	/*** TOP position */
47  	public static final int TOP		= 3;
48  	/*** CENTER position */
49  	public static final int CENTER	= 4;
50  	/*** FULL position */
51  	public static final int FULL	= 5;
52  
53  	/*
54  	 * The position of the component, can be any of the following values:
55  	 * LEFT, RIGHT, FULL.
56  	 */
57  	private int position;
58  
59  	/***
60  	 * horizontal alignment, can be any of the following values:
61  	 * LEFT, CENTER, RIGHT.
62  	 */
63      private int horizontalAlignment;
64  
65      /***
66       * Vertical alignment, can be any of the following values:
67       * TOP, CENTER, BOTTOM.
68       */
69      private int verticalAlignment;
70  
71      /***
72       * If true, the component fills the whole available horizontal space.
73       */
74      private boolean filled;
75  
76      /***
77  	 * Constructs a default FormConstraint which positions the 
78  	 * component over both columns and aligns the component 
79  	 * horizontally to the LEFT and vertically at the CENTER.
80       */
81      public FormConstraints() {
82  		this( FULL, LEFT, CENTER);
83      }
84  
85      /***
86       * Constructs a default FormConstraint which aligns the 
87  	 * component horizontally to the LEFT and vertically at 
88  	 * the CENTER and positions the component at the position 
89  	 * supplied.
90  	 *
91       * @param position the column position of the component.
92       */
93      public FormConstraints( int position) {
94      	this( position, LEFT, CENTER);
95      }
96  
97      /***
98       * Constructs a FormConstraint with the filled value set to 
99  	 * the value supplied. If the value is set to true, the alignment
100 	 * is no longer important.
101 	 *
102 	 * @param position the column position of the component.
103 	 * @param filled whether the component should fill the available 
104 	 * horizontal space.
105      */
106     public FormConstraints( int position, boolean filled) {
107 		this( position, LEFT, CENTER);
108 
109 	    this.filled = filled;
110     }
111 
112     /***
113      * Constructs a FormConstraint which aligns the component 
114 	 * horizontally with the value supplied and vertically at 
115 	 * the CENTER.
116 	 *
117 	 * @param position the column position of the component.
118 	 * @param horizontalAlignment the horizontal alignment type.
119      */
120     public FormConstraints( int position, int horizontalAlignment) {
121 		this( position, horizontalAlignment, CENTER);
122     }
123 
124     /***
125      * Constructs a FormConstraint which aligns the component 
126      * horizontally and vertically with the values supplied.
127      *
128      * @param position the column position of the component.
129      * @param horizontalAlignment the horizontal alignment type.
130      * @param verticalAlignment the vertical alignment type.
131      */
132     public FormConstraints( int position, int horizontalAlignment, int verticalAlignment) {
133 		this.position = position;
134 		this.horizontalAlignment = horizontalAlignment;
135 	    this.verticalAlignment = verticalAlignment;
136 		this.filled = false;
137     }
138     
139     /***
140      * Constructs a FormConstraint which copies its values 
141      * from the constraint supplied.
142      *
143      * @param constraint the form constraint to copy the values from.
144      */
145     public FormConstraints( FormConstraints constraint) {
146 	    this.horizontalAlignment = constraint.getHorizontalAlignment();
147 	    this.verticalAlignment = constraint.getVerticalAlignment();
148 	    this.filled = constraint.isFilled();
149 		this.position = constraint.getPosition();
150     }
151 
152     /***
153      * Sets wether the component should occupy the full available 
154 	 * horizontal space. The values for Horizontal and vertical alignment 
155 	 * are discarded when the component is filled.
156 	 *
157      * @param enabled enables/disables the filling of the full space.
158      */
159     public void setFilled( boolean enabled) {
160     	filled = enabled;
161     }
162 
163     /***
164      * Returns wether the component occupies the full available horizontal 
165 	 * space. The values for Horizontal and vertical alignment are discarded 
166 	 * when the component is filled.
167      *
168      * @return the filling of the available space.
169      */
170     public boolean isFilled() {
171     	return filled;
172     }
173 
174     /***
175      * Sets the horizontal alignment for the component.
176 	 * The alignment can be any of the following values:
177 	 * LEFT, for a left aligned component.
178 	 * CENTER, for a center aligned component.
179 	 * RIGHT, for a right aligned component.
180      *
181      * @param alignment the horizontal alignment.
182      */
183     public void setHorizontalAlignment( int alignment) {
184 		horizontalAlignment = alignment;
185     }
186 
187     /***
188      * Returns the horizontal alignment for the component.
189 	 * @see #setHorizontalAlignment
190      *
191      * @return the horizontal alignment.
192      */
193     public int getHorizontalAlignment() {
194 		return horizontalAlignment;
195     }
196 
197     /***
198      * Sets the vertical alignment for the component.
199 	 * The alignment can be any of the following values:
200 	 * TOP, for a top aligned component.
201 	 * CENTER, for a center aligned component.
202 	 * BOTTOM, for a bottom aligned component.
203      *
204      * @param alignment the vertical alignment.
205      */
206     public void setVerticalAlignment( int alignment) {
207 		verticalAlignment = alignment;
208     }
209 
210     /***
211      * Returns the vertical alignment for the component.
212 	 * @see #setVerticalAlignment
213      *
214      * @return the vertical alignment.
215      */
216     public int getVerticalAlignment() {
217 		return verticalAlignment;
218     }
219 
220     /***
221      * Sets the column position of the component;
222 	 * LEFT, for a component in the left column.
223 	 * RIGHT, for a component in the right column.
224 	 * FULL, for a component that occupies both columns.
225      *
226      * @param position the column position of the component.
227      */
228     public void setPosition( int position) {
229 		this.position = position;
230     }
231 
232     /***
233      * Returns the column position of the component.
234 	 * @see #setPosition
235      *
236      * @return the column position.
237      */
238     public int getPosition() {
239 		return position;
240     }
241 
242     /***
243      * Finds out if the values in this FormConstraint are equal to 
244 	 * the supplied object.
245      *
246      * @param object the object to test.
247      *
248      * @return true when equal.
249      */
250     public boolean equals( Object object) {
251 		boolean equal = true;
252 		
253 		if ( this != object) {
254 			if ( object instanceof FormConstraints) {
255 				FormConstraints constraint = (FormConstraints) object;
256 
257 				if ( constraint.isFilled() != filled) {
258 					equal = false;
259 				} else if ( constraint.getPosition() != position) {
260 					equal = false;
261 				} else if ( constraint.getHorizontalAlignment() != horizontalAlignment) {
262 					equal = false;
263 				} else if ( constraint.getVerticalAlignment() != verticalAlignment) {
264 					equal = false;
265 				} 
266 			} else {
267 				equal = false;
268 			}
269 		}
270 
271 		return equal;
272     }
