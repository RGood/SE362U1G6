30  package org.bounce.viewer.xml;
31  
32  import java.util.ArrayList;
33  import java.util.List;
34  
35  class StyledElement {
36  	private List<StyledString> strings = null;
37  	
38  	StyledElement() {
39  		strings = new ArrayList<StyledString>();
40  	}
41  
42  	void addString(StyledString string) {
43  		strings.add(string);
44  	}
45  	
46  	int length() {
47  		int result = 0;
48  		
49  		for (StyledString string : strings) {
50  			result += string.getText().length();
51  		}
52  		
53  		return result;
54  	}
55  	
56  	List<StyledString> getStrings() {
57  		return strings;
58  	}
59  }
