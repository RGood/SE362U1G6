83  	public int getEventType() {
84  		if (tagScanner == TAG_SCANNER) {
85  			if (TAG_SCANNER.scanner == ELEMENT_START_SCANNER) {
86  				// if (ELEMENT_START_TAG_SCANNER.scanner == ATTRIBUTE_SCANNER) {
87  				// return TYPE.ATTRIBUTE;
88  				// }
89  
90  				if (ELEMENT_START_SCANNER.scanner == TAG_END_SCANNER && TAG_END_SCANNER.emptyElement) {
91  					return XMLEvent.END_ELEMENT;
92  				}
93  
94  				return XMLEvent.START_ELEMENT;
95  			} else if (TAG_SCANNER.scanner == ELEMENT_END_SCANNER) {
96  				return XMLEvent.END_ELEMENT;
97  			} else if (TAG_SCANNER.scanner == COMMENT_SCANNER) {
98  				return XMLEvent.COMMENT;
99  			} else if (TAG_SCANNER.scanner == CDATA_SCANNER) {
100 				return XMLEvent.CDATA;
101 			}
102 		} else if (tagScanner == CONTENT_SCANNER) {
103 			return XMLEvent.CHARACTERS;
104 		}
105 		
106 		if (getStartOffset() == 0) {
107 			return XMLEvent.START_DOCUMENT;
108 		}
109 		
110 		return XMLEvent.END_DOCUMENT;
111 	}

113 	public int getNextTag() throws IOException {
114 		while (true) {
115 			scan();
116 			
117 			if (token == XMLStyleConstants.ELEMENT_NAME) {
118 				return getEventType();
119 			} else if (tagScanner == TAG_SCANNER && TAG_SCANNER.scanner == ELEMENT_START_SCANNER && ELEMENT_START_SCANNER.scanner == TAG_END_SCANNER && TAG_END_SCANNER.emptyElement) {
120 				return XMLEvent.END_ELEMENT;
121 			} else if (in.getLastChar() == -1) {
122 				return getEventType();
123 			}
124 		}
125 	}

143 	/***
144 	 * Scans the Xml Stream for XML specific tokens.
145 	 * 
146 	 * @return the last location.
147 	 * 
148 	 * @throws IOException
149 	 */
150 	public long scan() throws IOException {
151 		int character = in.getLastChar();
152 
153 		if (error && (character == '<' || character == -1)) {
154 			tagScanner = null;
155 			token = null;
156 		}
157 
158 		error = false;
159 
160 		if (tagScanner != null && tagScanner.isFinished()) {
161 			tagScanner = null;
162 			token = null;
163 		}
164 
165 		long l = pos;
166 		pos = in.pos;
167 
168 		while (true) {
169 			if (tagScanner != null) {
170 				token = tagScanner.scan(in);
171 
172 				character = in.getLastChar();
173 
174 				if (getEndOffset() > getStartOffset() || character == -1) {
175 					break;
176 				} else if (character == '<') {
177 					character = in.read();
178 				}
179 			} else if (character == '<') {
180 				character = in.read();
181 
182 				tagScanner = TAG_SCANNER;
183 				tagScanner.reset();
184 
185 			} else if (isContent(character) || character == '&') {
186 				tagScanner = CONTENT_SCANNER;
187 				tagScanner.reset();
188 			} else {
189 				token = null;
190 				error = true;
191 				break;
192 			}
193 		}
194 		
195 		if (error) {
196 			if (in.getLastChar() == -1 && getStartOffset() == getEndOffset()) {
197 				token = null;
198 				tagScanner = null;
199 			}
200 		}
201 		
202 		return l;
203 	}

205 	/***
206 	 * A scanner for anything starting with a ' <'.
207 	 */
208 	private class TagScanner extends Scanner {
209 		private Scanner scanner = null;
210 
211 		/***
212 		 * returns whether this scanner has finished scanning all it was
213 		 * supposed to scan.
214 		 * 
215 		 * @return true when the scanner is finished.
216 		 */
217 		public boolean isFinished() {
218 			return scanner != null && scanner.isFinished();
219 		}
220 
221 		/***
222 		 * @see Scanner#scan(DocumentInputReader)
223 		 */
224 		public String scan(DocumentInputReader in) throws IOException {
225 			if (scanner != null) {
226 				if (scanner.isFinished()) {
227 					scanner = null;
228 				} else {
229 					return scanner.scan(in);
230 				}
231 			}
232 
233 			int character = in.getLastChar();
234 
235 			if (character == 33) { // '!'
236 				character = in.read();
237 				if (character == 45) { // '-'
238 					character = in.read();
239 					if (character == 45) { // '-'
240 						character = in.read();
241 						scanner = COMMENT_SCANNER;
242 						scanner.reset();
243 						return XMLStyleConstants.COMMENT;
244 					}
245 
246 				}
247 
248 				if (character == 91) { // '['
249 					character = in.read();
250 					if (character == 67) { // 'C'
251 						character = in.read();
252 						if (character == 68) { // 'D'
253 							character = in.read();
254 							if (character == 65) { // 'A'
255 								character = in.read();
256 								if (character == 84) { // 'T'
257 									character = in.read();
258 									if (character == 65) { // 'A'
259 										character = in.read();
260 										if (character == 91) { // '['
261 											character = in.read();
262 											scanner = CDATA_SCANNER;
263 											scanner.reset();
264 											return XMLStyleConstants.CDATA;
265 										}
266 									}
267 								}
268 							}
269 						}
270 					}
271 				}
272 
273 				if (scanner == null) {
274 					scanner = ENTITY_TAG_SCANNER;
275 				}
276 
277 				scanner.reset();
278 				return XMLStyleConstants.SPECIAL;
279 
280 			} else if (character == '?') { // '?'
281 				character = in.read();
282 				scanner = ENTITY_TAG_SCANNER;
283 				scanner.reset();
284 
285 				return XMLStyleConstants.SPECIAL;
286 
287 			} else if (character == '/') { // '/'
288 				character = in.read();
289 				scanner = ELEMENT_END_SCANNER;
290 				scanner.reset();
291 
292 				return XMLStyleConstants.SPECIAL;
293 
294 			} else if (character == '>') { // '>'
295 				character = in.read();
296 				finished();
297 				return XMLStyleConstants.SPECIAL;
298 
299 			} else if (character == '<') { // '>'
300 				scanner = ELEMENT_START_SCANNER; // for the show
301 				scanner.reset();
302 
303 				error = true;
304 				finished();
305 				return XMLStyleConstants.SPECIAL;
306 			} else {
307 				scanner = ELEMENT_START_SCANNER;
308 				scanner.reset();
309 
310 				return XMLStyleConstants.SPECIAL;
311 			}
312 		}
313 
314 		/***
315 		 * @see Scanner#reset()
316 		 */
317 		public void reset() {
318 			super.reset();
319 			scanner = null;
320 		}
321 	}

344 
345 				default:
346 					character = in.read();
347 					break;
348 
349 				}
350 			}

388 						}
389 					}
390 					break;
391 
392 				default:
393 					character = in.read();
394 					break;
395 
396 				}

456 	 * Scans an element start tag ' <xxx:xxxx yyy:yyyy="yyyyy"
457 	 * xmlns:hsshhs="sffsfsf">'.

459 	private class ElementEndScanner extends Scanner {
460 		private Scanner scanner = ELEMENT_NAME_SCANNER;
461 
462 		/***
463 		 * @see Scanner#scan(DocumentInputReader)
464 		 */
465 		public String scan(DocumentInputReader in) throws IOException {
466 			String token = scanner.scan(in);
467 
468 			if (scanner.isFinished()) {
469 				if (scanner == TAG_END_SCANNER) {
470 					finished();
471 				} else {
472 					int character = in.getLastChar();
473 					
474 					if (character == '>') {
475 						scanner = TAG_END_SCANNER;
476 						scanner.reset();
477 					} else if (isSpace((char)character)) {
478 						scanner = WHITESPACE_SCANNER;
479 						scanner.reset();
480 					} else {
481 						error = true;
482 
483 						if (character == '<' || character == -1) {
484 							finished();
485 						} else {
486 							character = in.read();
487 						}
488 					}
489 				}
490 			}
491 			
492 			return token;
493 		}
494 
495 		/***
496 		 * @see Scanner#reset()
497 		 */
498 		public void reset() {
499 			super.reset();
500 			scanner = ELEMENT_NAME_SCANNER;
501 			scanner.reset();
502 		}
503 	}

571 	/***
572 	 * Scans an element name ' <xxx:xxxx'.
573 	 */
574 	private class ElementNameScanner extends Scanner {
575 		private boolean prefix = false;
576 		private boolean first = true;
577 		private boolean nameStart = false;
578 
579 		/***
580 		 * @see Scanner#scan(DocumentInputReader)
581 		 */
582 		public String scan(DocumentInputReader in) throws IOException {
583 
584 			int character = in.getLastChar();
585 			
586 			if (first) {
587 				first = false;
588 				
589 				if (!isNameStart(character)) {
590 					error = true;
591 
592 					if (character == '<' || character == -1) {
593 						finished();
594 						return XMLStyleConstants.ELEMENT_NAME;
595 					}
596 				}
597 
598 				character = in.read();
599 			}
600 			
601 			do {
602 				if (nameStart) {
603 					nameStart = false;
604 
605 					if (isNameStart(character)) {
606 						character = in.read();
607 					} else {
608 						error = true;
609 
610 						if (character == '<' || character == -1) {
611 							finished();
612 
613 							return XMLStyleConstants.ELEMENT_NAME;
614 						} else {
615 							character = in.read();
616 						}
617 					}
618 				} else if (character == ':') {
619 					if (prefix) {
620 						character = in.read();
621 						nameStart = true;
622 						return XMLStyleConstants.SPECIAL;
623 					}
624 
625 					prefix = true;
626 					return XMLStyleConstants.ELEMENT_PREFIX;
627 				} else if (isName(character)) {
628 					character = in.read();
629 				} else if (isSpace(character) || character == '/' || character == '>') {
630 					finished();
631 					return XMLStyleConstants.ELEMENT_NAME;
632 				} else {
633 					error = true;
634 
635 					if (character == '<' || character == -1) {
636 						finished();
637 						return XMLStyleConstants.ELEMENT_NAME;
638 					} else {
639 						character = in.read();
640 					}
641 				}
642 			} while (true);
643 		}
644 
645 		/***
646 		 * @see Scanner#reset()
647 		 */
648 		public void reset() {
649 			super.reset();
650 			prefix = false;
651 			first = true;
652 			nameStart = false;
653 		}
654 	}

656 	private class TagEndScanner extends Scanner {
657 		boolean emptyElement = false;
658 		
659 		/***
660 		 * @see Scanner#scan(DocumentInputReader)
661 		 */
662 		public String scan(DocumentInputReader in) throws IOException {
663 			int character = in.getLastChar();
664 
665 			do {
666 				if (character == '/') {
667 					emptyElement = true;
668 					character = in.read();
669 				} else if (character == '>') {
670 					character = in.read();
671 					finished();
672 					return XMLStyleConstants.SPECIAL;
673 				} else {
674 					if (character == '<' || character == -1) {
675 						finished();
676 						return XMLStyleConstants.SPECIAL;
677 					} else {
678 						character = in.read();
679 					}
680 
681 					error = true;
682 				}
683 			} while (true);
684 		}
685 
686 		/***
687 		 * @see Scanner#reset()
688 		 */
689 		public void reset() {
690 			super.reset();
691 			emptyElement = false;
692 		}
693 	}

695 	/***
696 	 * Scans an elements attribute 'xxx:xxxx="hhhh"' or 'xmlns:xxxx="hhhh"'.
697 	 */
698 	private class AttributeScanner extends Scanner {
699 		private Scanner scanner = ATTRIBUTE_NAME_SCANNER;
700 		boolean foundEquals = false;
701 
702 		/***
703 		 * @see Scanner#scan(DocumentInputReader)
704 		 */
705 		public String scan(DocumentInputReader in) throws IOException {
706 			String token = null;
707 			int ch = in.getLastChar();
708 			
709 			if (ch == '=') {
710 				in.read();
711 				foundEquals = true;
712 				
713 				token = XMLStyleConstants.SPECIAL;
714 			} else {
715 				token = scanner.scan(in);
716 			}
717 
718 			if (scanner.isFinished()) {
719 				if (scanner == ATTRIBUTE_VALUE_SCANNER || scanner == NAMESPACE_VALUE_SCANNER) {
720 					finished();
721 				} else {
722 					ch = in.getLastChar();
723 					
724 					if (isSpace((char)ch)) {
725 						scanner = WHITESPACE_SCANNER;
726 						scanner.reset();
727 					} else if (ch == '\'' || ch == '"') {
728 						if (ATTRIBUTE_NAME_SCANNER.namespace) {
