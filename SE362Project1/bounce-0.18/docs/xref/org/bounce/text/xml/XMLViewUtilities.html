39  /***
40   * The XML View uses the XML scanner to determine the style (font, color) of the
41   * text that it renders.
42   * <p>
43   * <b>Note: </b> The XML Editor package is based on the JavaEditorKit example as
44   * described in the article <i>'Customizing a Text Editor' </i> by <b>Timothy
45   * Prinzing </b>. See:
46   * http://java.sun.com/products/jfc/tsc/articles/text/editor_kit/
47   * </p>

49   * @author Edwin Dankert <edankert@gmail.com>
50   * @version $Revision: 1.5 $, $Date: 2009/01/22 22:14:59 $
51   */
52  class XMLViewUtilities {
53  	// Update the scanner to point to the '<' begin token.
54  	public static void updateScanner(SyntaxHighlightingScanner scanner, Document doc, int start, int end) {
55  		try {
56  			if (!scanner.isValid()) {
57  				scanner.setRange(getTagEnd(doc, start), end);
58  				scanner.setValid(true);
59  			}
60  
61  			while (scanner.getEndOffset() <= start && end > scanner.getEndOffset()) {
62  				scanner.scan();
63  			}
64  		} catch (IOException e) {
65  			// can't adjust scanner... calling logic
66  			// will simply render the remaining text.
67  			// e.printStackTrace();
68  		}
69  	}
70  
71  	// Return the end position of the current tag.
72  	private static int getTagEnd(Document doc, int p) {
73  		int elementEnd = 0;
74  
75  		if (p > 0) {
76  			try {
77  				int index;
78  
79  				String s = doc.getText(0, p);
80  				int cdataStart = s.lastIndexOf("<![CDATA[");
81  				int cdataEnd = s.lastIndexOf("]]>");
82  				int commentStart = s.lastIndexOf("<!--");
83  				int commentEnd = s.lastIndexOf("-->");
84  
85  				if (cdataStart > 0 && cdataStart > cdataEnd) {
86  					index = s.lastIndexOf(">", cdataStart);
87  				} else if (commentStart > 0 && commentStart > commentEnd) {
88  					index = s.lastIndexOf(">", commentStart);
89  				} else {
90  					index = s.lastIndexOf(">");
91  				}
92  
93  				if (index != -1)
94  					elementEnd = index;
95  			} catch (BadLocationException bl) {
96  				// empty
97  			}
98  		}
99  
100 		return elementEnd;
101 	}
