13  /***
14   * BrowserLauncher is a class that provides one static method, openURL, which opens the default
15   * web browser for the current user of the system to the given URL.  It may support other
16   * protocols depending on the system -- mailto, ftp, etc. -- but that has not been rigorously
17   * tested and is not guaranteed to work.
18   * <p>
19   * Yes, this is platform-specific code, and yes, it may rely on classes on certain platforms
20   * that are not part of the standard JDK.  What we're trying to do, though, is to take something
21   * that's frequently desirable but inherently platform-specific -- opening a default browser --
22   * and allow programmers (you, for example) to do so without worrying about dropping into native
23   * code or doing anything else similarly evil.
24   * <p>
25   * Anyway, this code is completely in Java and will run on all JDK 1.1-compliant systems without
26   * modification or a need for additional libraries.  All classes that are required on certain
27   * platforms to allow this to run are dynamically loaded at runtime via reflection and, if not
28   * found, will not cause this to do anything other than returning an error when opening the
29   * browser.
30   * <p>
31   * There are certain system requirements for this class, as it's running through Runtime.exec(),
32   * which is Java's way of making a native system call.  Currently, this requires that a Macintosh
33   * have a Finder which supports the GURL event, which is true for Mac OS 8.0 and 8.1 systems that
34   * have the Internet Scripting AppleScript dictionary installed in the Scripting Additions folder
35   * in the Extensions folder (which is installed by default as far as I know under Mac OS 8.0 and
36   * 8.1), and for all Mac OS 8.5 and later systems.  On Windows, it only runs under Win32 systems
37   * (Windows 95, 98, and NT 4.0, as well as later versions of all).  On other systems, this drops
38   * back from the inherently platform-sensitive concept of a default browser and simply attempts
39   * to launch Netscape via a shell command.
40   * <p>
41   * This code is Copyright 1999-2002 by Eric Albert (ejalbert@cs.stanford.edu) and may be
42   * redistributed or modified in any form without restrictions as long as the portion of this
43   * comment from this paragraph through the end of the comment is not removed.  The author
44   * requests that he be notified of any application, applet, or other binary that makes use of
45   * this code, but that's more out of curiosity than anything and is not required.  This software
46   * includes no warranty.  The author is not repsonsible for any loss of data or functionality
47   * or any adverse or unexpected effects of using this software.

192 	/***
193 	 * An initialization block that determines the operating system and loads the necessary
194 	 * runtime data.
195 	 */
196 	static {
197 		loadedWithoutErrors = true;
198 		String osName = System.getProperty("os.name");
199 		if (osName.startsWith("Mac OS")) {
200 			String mrjVersion = System.getProperty("mrj.version");
201 			String majorMRJVersion = mrjVersion.substring(0, 3);
202 			try {
203 				double version = Double.valueOf(majorMRJVersion).doubleValue();
204 				if (version == 2) {
205 					jvm = MRJ_2_0;
206 				} else if (version >= 2.1 && version < 3) {
207 					// Assume that all post-2.1 2.x versions of MRJ work the same. MRJ 2.1 actually
208 					// works via Runtime.exec() and 2.2 supports that but has an openURL() method
209 					// as well that we don't use because the Runtime.exec() method works fine.
210 					jvm = MRJ_2_1;
211 				} else if (version == 3.0) {
212 					jvm = MRJ_3_0;
213 				} else if (version >= 3.1) {
214 					// Assume that all 3.1 and later versions of MRJ work the same.
215 					jvm = MRJ_3_1;
216 				} else {
217 					loadedWithoutErrors = false;
218 					errorMessage = "Unsupported MRJ version: " + version;
219 				}
220 			} catch (NumberFormatException nfe) {
221 				loadedWithoutErrors = false;
222 				errorMessage = "Invalid MRJ version: " + mrjVersion;
223 			}
224 		} else if (osName.startsWith("Windows")) {
225 			if (osName.indexOf("9") != -1 || osName.indexOf("Me") != -1) {
226 				jvm = WINDOWS_9x;
227 			} else {
228 				jvm = WINDOWS_NT;
229 			}
230 		} else {
231 			jvm = OTHER;
232 		}
233 		
234 		if (loadedWithoutErrors) {	// if we haven't hit any errors yet
235 			loadedWithoutErrors = loadClasses();
236 		}
237 	}

363 	/***
364 	 * Attempts to locate the default web browser on the local system.  Caches the result so it
365 	 * only locates the browser once for each use of this class per JVM instance.
366 	 * @return The browser for the system.  Note that this may not be what you would consider
367 	 *			to be a standard web browser; instead, it's the application that gets called to
368 	 *			open the default web browser.  In some cases this will be a non-String object
369 	 *			that provides the means of calling the default browser.
370 	 */
371 	private static Object locateBrowser() {
372 		if (browser != null) {
373 			return browser;
374 		}
375 		switch (jvm) {
376 			case MRJ_2_0:
377 				try {
378 					Integer finderCreatorCode = (Integer) makeOSType.invoke(null, new Object[] { FINDER_CREATOR });
379 					Object aeTarget = aeTargetConstructor.newInstance(new Object[] { finderCreatorCode });
380 					Integer gurlType = (Integer) makeOSType.invoke(null, new Object[] { GURL_EVENT });
381 					Object appleEvent = appleEventConstructor.newInstance(new Object[] { gurlType, gurlType, aeTarget, kAutoGenerateReturnID, kAnyTransactionID });
382 					// Don't set browser = appleEvent because then the next time we call
383 					// locateBrowser(), we'll get the same AppleEvent, to which we'll already have
384 					// added the relevant parameter. Instead, regenerate the AppleEvent every time.
385 					// There's probably a way to do this better; if any has any ideas, please let
386 					// me know.
387 					return appleEvent;
388 				} catch (IllegalAccessException iae) {
389 					browser = null;
390 					errorMessage = iae.getMessage();
391 					return browser;
392 				} catch (InstantiationException ie) {
393 					browser = null;
394 					errorMessage = ie.getMessage();
395 					return browser;
396 				} catch (InvocationTargetException ite) {
397 					browser = null;
398 					errorMessage = ite.getMessage();
399 					return browser;
400 				}
401 			case MRJ_2_1:
402 				File systemFolder;
403 				try {
404 					systemFolder = (File) findFolder.invoke(null, new Object[] { kSystemFolderType });
405 				} catch (IllegalArgumentException iare) {
406 					browser = null;
407 					errorMessage = iare.getMessage();
408 					return browser;
409 				} catch (IllegalAccessException iae) {
410 					browser = null;
411 					errorMessage = iae.getMessage();
412 					return browser;
413 				} catch (InvocationTargetException ite) {
414 					browser = null;
415 					errorMessage = ite.getTargetException().getClass() + ": " + ite.getTargetException().getMessage();
416 					return browser;
417 				}
418 				String[] systemFolderFiles = systemFolder.list();
419 				// Avoid a FilenameFilter because that can't be stopped mid-list
420 				for(int i = 0; i < systemFolderFiles.length; i++) {
421 					try {
422 						File file = new File(systemFolder, systemFolderFiles[i]);
423 						if (!file.isFile()) {
424 							continue;
425 						}
426 						// We're looking for a file with a creator code of 'MACS' and
427 						// a type of 'FNDR'.  Only requiring the type results in non-Finder
428 						// applications being picked up on certain Mac OS 9 systems,
429 						// especially German ones, and sending a GURL event to those
430 						// applications results in a logout under Multiple Users.
431 						Object fileType = getFileType.invoke(null, new Object[] { file });
432 						if (FINDER_TYPE.equals(fileType.toString())) {
433 							Object fileCreator = getFileCreator.invoke(null, new Object[] { file });
434 							if (FINDER_CREATOR.equals(fileCreator.toString())) {
435 								browser = file.toString();	// Actually the Finder, but that's OK
436 								return browser;
437 							}
438 						}
439 					} catch (IllegalArgumentException iare) {
440 						errorMessage = iare.getMessage();
441 						return null;
442 					} catch (IllegalAccessException iae) {
443 						browser = null;
444 						errorMessage = iae.getMessage();
445 						return browser;
446 					} catch (InvocationTargetException ite) {
447 						browser = null;
448 						errorMessage = ite.getTargetException().getClass() + ": " + ite.getTargetException().getMessage();
449 						return browser;
450 					}
451 				}
452 				browser = null;
453 				break;
454 			case MRJ_3_0:
455 			case MRJ_3_1:
456 				browser = "";	// Return something non-null
457 				break;
458 			case WINDOWS_NT:
459 				browser = "cmd.exe";
460 				break;
461 			case WINDOWS_9x:
462 				browser = "command.com";
463 				break;
464 			case OTHER:
465 			default:
466 				// On systems other than Windows and the Mac, we try via a rather Unix-
467 				// specific hack to read the BROWSER environment variable
468 				// <http://tuxedo.org/~esr/BROWSER/>. If we can't read that variable or
469 				// it isn't set, we use Netscape.
470 
471 				browser = getEnvironmentBrowser();
472 				if (browser == null) {
473 					browser = "netscape";
474 				}
475 				break;
476 		}
477 		return browser;
478 	}

480 	/***
481 	 * Attempts to open the default web browser to the given URL.
482 	 * @param url The URL to open
483 	 * @throws IOException If the web browser could not be located or does not run
484 	 */
485 	public static void openURL(String url) throws IOException {
486 		if (!loadedWithoutErrors) {
487 			throw new IOException("Exception in finding browser: " + errorMessage);
488 		}
489 		Object browser = locateBrowser();
490 		if (browser == null) {
491 			throw new IOException("Unable to locate browser: " + errorMessage);
492 		}
493 		
494 		switch (jvm) {
495 			case MRJ_2_0:
496 				Object aeDesc = null;
497 				try {
498 					aeDesc = aeDescConstructor.newInstance(new Object[] { url });
499 					putParameter.invoke(browser, new Object[] { keyDirectObject, aeDesc });
500 					sendNoReply.invoke(browser, new Object[] { });
501 				} catch (InvocationTargetException ite) {
502 					throw new IOException("InvocationTargetException while creating AEDesc: " + ite.getMessage());
503 				} catch (IllegalAccessException iae) {
504 					throw new IOException("IllegalAccessException while building AppleEvent: " + iae.getMessage());
505 				} catch (InstantiationException ie) {
506 					throw new IOException("InstantiationException while creating AEDesc: " + ie.getMessage());
507 				} finally {
508 					aeDesc = null;	// Encourage it to get disposed if it was created
509 					browser = null;	// Ditto
510 				}
511 				break;
512 			case MRJ_2_1:
513 				Runtime.getRuntime().exec(new String[] { (String) browser, url } );
514 				break;
515 			case MRJ_3_0:
516 				int[] instance = new int[1];
517 				int result = ICStart(instance, 0);
518 				if (result == 0) {
519 					int[] selectionStart = new int[] { 0 };
520 					byte[] urlBytes = url.getBytes();
521 					int[] selectionEnd = new int[] { urlBytes.length };
522 					result = ICLaunchURL(instance[0], new byte[] { 0 }, urlBytes,
523 											urlBytes.length, selectionStart,
524 											selectionEnd);
525 					if (result == 0) {
526 						// Ignore the return value; the URL was launched successfully
527 						// regardless of what happens here.
528 						ICStop(instance);
529 					} else {
530 						throw new IOException("Unable to launch URL: " + result);
531 					}
532 				} else {
533 					throw new IOException("Unable to create an Internet Config instance: " + result);
534 				}
535 				break;
536 			case MRJ_3_1:
537 				try {
538 					openURL.invoke(null, new Object[] { url });
539 				} catch (InvocationTargetException ite) {
540 					throw new IOException("InvocationTargetException while calling openURL: " + ite.getMessage());
541 				} catch (IllegalAccessException iae) {
542 					throw new IOException("IllegalAccessException while calling openURL: " + iae.getMessage());
543 				}
544 				break;
545 		    case WINDOWS_NT:
546 		    case WINDOWS_9x:
547 		    	// Add quotes around the URL to allow ampersands and other special
548 		    	// characters to work.
549 		    	String[] arguments;
550 		    	if (jvm == WINDOWS_9x) {
551 		    		arguments = new String[] { (String) browser,
552 		    							  FIRST_WINDOWS_PARAMETER,
553 		    							  SECOND_WINDOWS_PARAMETER,
554 		    							  null };
555 		    	} else {
556 		    		arguments = new String[] { (String) browser,
557 		    							  FIRST_WINDOWS_PARAMETER,
558 		    							  SECOND_WINDOWS_PARAMETER,
559 		    							  THIRD_WINDOWS_PARAMETER,
560 		    							  null };
561 		    	}
562 		    	arguments[arguments.length - 1] = '"' + url + '"';
563 				Process process = Runtime.getRuntime().exec(arguments);
564 				// This avoids a memory leak on some versions of Java on Windows.
565 				// That's hinted at in <http://developer.java.sun.com/developer/qow/archive/68/>.
566 				try {
567 					process.waitFor();
568 					process.exitValue();
569 				} catch (InterruptedException ie) {
570 					throw new IOException("InterruptedException while launching browser: " + ie.getMessage());
571 				}
572 				break;
573 			case OTHER:
574 				// Assume that we're on Unix and that Netscape is installed
575 				
576 				// Attempt to open the URL in a currently running session of Netscape
577 				process = Runtime.getRuntime().exec(new String[] { (String) browser,
578 													NETSCAPE_REMOTE_PARAMETER,
579 													NETSCAPE_OPEN_PARAMETER_START +
580 													url +
581 													NETSCAPE_OPEN_PARAMETER_END });
582 				try {
583 					int exitCode = process.waitFor();
584 					if (exitCode != 0) {	// if Netscape was not open
585 						Runtime.getRuntime().exec(new String[] { (String) browser, url });
586 					}
587 				} catch (InterruptedException ie) {
588 					throw new IOException("InterruptedException while launching browser: " + ie.getMessage());
589 				}
590 				break;
591 			default:
592 				// This should never occur, but if it does, we'll try the simplest thing possible
593 				Runtime.getRuntime().exec(new String[] { (String) browser, url });
594 				break;
595 		}
596 	}

598 	/***
599 	 * Tries to read the BROWSER environment variable, which should be set to the absolute
600 	 * path of the user's preferred web browser as proposed at <http://tuxedo.org/~esr/BROWSER/>.
601 	 * @return The value of the BROWSER environment variable, or null if the variable does
602 	 *		   not exist or can't be read.
603 	 */
604 	private static String getEnvironmentBrowser() {
605 		String browser = System.getProperty("org.bounce.browser", null);
606 		
607 		if ( browser == null) {
608 			try {
609 				String[] echoParams = { "/bin/sh", "-c", "echo $BROWSER" };
610 				Process echoProcess = Runtime.getRuntime().exec(echoParams);
611 				InputStream echoStream = echoProcess.getInputStream();
612 		
613 				BufferedReader echoReader = null;
614 			
615 				try {
616 					echoReader = new BufferedReader(new InputStreamReader(echoStream));
617 		 			browser = echoReader.readLine();
618 				} finally {
619 					if (echoReader != null) {
620 			 			echoReader.close();
621 					}
622 				}
623 				
624 				browser = browser.trim();
625 				if ( browser.length() == 0) {
626 					browser = null;
627 				}
628 			} catch (Throwable t) {
629 				// If anything goes wrong, we'll return null.
630 			}
631 		}
632 
633 		return browser;
634 	}
