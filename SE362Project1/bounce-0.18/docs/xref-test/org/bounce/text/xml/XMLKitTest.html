30  package org.bounce.text.xml;
31  
32  import java.awt.BorderLayout;
33  import java.awt.Color;
34  import java.awt.Font;
35  
36  import javax.swing.JEditorPane;
37  import javax.swing.JFrame;
38  import javax.swing.JPanel;
39  import javax.swing.JScrollPane;
40  import javax.swing.text.PlainDocument;
41  
42  import org.bounce.com.sun.syndication.io.XmlReader;
43  import org.bounce.text.LineNumberMargin;
44  import org.bounce.text.ScrollableEditorPanel;
45  
46  /***
47   * Simple wrapper around JEditorPane to browse java text using the XMLEditorKit
48   * plug-in.
49   * 
50   * java XmlKitTest filename
51   */
52  public class XMLKitTest {
53  	private static JEditorPane editor = null;
54  
55  	/***
56  	 * Main method...
57  	 * 
58  	 * @param args
59  	 */
60  	public static void main(String[] args) {
61  //		if (args.length != 1) {
62  //			System.err.println("need filename argument");
63  //			System.exit(1);
64  //		}
65  
66  		try {
67  			editor = new JEditorPane();
68  
69  			// Instantiate a XMLEditorKit with wrapping enabled.
70  			XMLEditorKit kit = new XMLEditorKit();
71  
72  			editor.setEditorKit(kit);
73  
74  			// Set the font style.
75  			editor.setFont(new Font("Courier", Font.PLAIN, 12));
76  
77  			// Set the tab size
78  			editor.getDocument().putProperty(PlainDocument.tabSizeAttribute, new Integer(4));
79  
80  			// Enable auto indentation.
81  			kit.setAutoIndentation(true);
82  
83  			// Enable error highlighting.
84  			editor.getDocument().putProperty(XMLEditorKit.ERROR_HIGHLIGHTING_ATTRIBUTE, new Boolean(true));
85  
86  			// Enable tag completion.
87  			kit.setTagCompletion(true);
88  
89  			// Set a style
90  			kit.setStyle(XMLStyleConstants.ATTRIBUTE_NAME, new Color(255, 0, 0), Font.BOLD);
91  
92  			// Put the editor in a panel that will force it to resize, when a
93  			// different view is choosen.
94  			ScrollableEditorPanel editorPanel = new ScrollableEditorPanel(editor);
95  
96  			JScrollPane scroller = new JScrollPane(editorPanel);
97  
98  			// Add the number margin and folding margin as a Row Header View
99  			JPanel rowHeader = new JPanel(new BorderLayout());
100 			rowHeader.add(new XMLFoldingMargin(editor), BorderLayout.EAST);
101 			rowHeader.add(new LineNumberMargin(editor), BorderLayout.WEST);
102 			scroller.setRowHeaderView(rowHeader);
103 
104 			editor.read(new XmlReader(XMLKitTest.class.getResourceAsStream("/test.xml")), null);
105 
106 			JFrame f = new JFrame("XmlEditorKitTest: " + "demo");
107 			f.getContentPane().setLayout(new BorderLayout());
108 			f.getContentPane().add(scroller, BorderLayout.CENTER);
109 
110 			f.setSize(600, 600);
111 			f.setVisible(true);
112 		} catch (Throwable e) {
113 			e.printStackTrace();
114 			System.exit(1);
115 		}
116 	}
117 }
